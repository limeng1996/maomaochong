        function switchTab(tab) {

            document.getElementById('tab-orders').className = tab === 'orders' ? 'flex-1 py-2 text-sm font-medium rounded-lg bg-white text-stone-800 shadow-sm transition-all' : 'flex-1 py-2 text-sm font-medium rounded-lg text-stone-500 transition-all';

            document.getElementById('panel-orders').classList.toggle('hidden', tab !== 'orders');

            document.getElementById('panel-settings').classList.toggle('hidden', tab !== 'settings');

            if (tab === 'orders') renderPending();

        }

                </div>

            `).join('');

        async function finishOrder(id) {

            if (!supabase) return;

            await supabase.from('orders').update({ status: 'done' }).eq('id', id);

            pendingOrders = pendingOrders.filter(o => o.id !== id);

            renderPending();

        }


        function upQrcode(input, type) {

            const f = input.files[0]; if (!f) return;

            const r = new FileReader();

            r.onload = (e) => {

                const img = new Image();

                img.onload = () => {

                    const canvas = document.createElement('canvas');

                    const scale = 300 / img.width;

                    canvas.width = 300; canvas.height = img.height * scale;

                    canvas.getContext('2d').drawImage(img, 0, 0, canvas.width, canvas.height);

                    const data = canvas.toDataURL('image/jpeg', 0.6);

                    if (type === 'wechat') { configData.wechat_qrcode = data; document.getElementById('p-wechat').innerHTML = `<img src="${data}" class="w-full h-full object-cover">`; }

                    else { configData.alipay_qrcode = data; document.getElementById('p-alipay').innerHTML = `<img src="${data}" class="w-full h-full object-cover">`; }

                };

                img.src = e.target.result;

            };

            r.readAsDataURL(f);

        }


        async function saveSettings() {

            if (!supabase) { alert('数据库未配置'); return; }

            const row = {

                shop_name: document.getElementById('s-name').value || '移动咖啡车',

                shop_notice: document.getElementById('s-notice').value || '新鲜烘焙 · 现场制作',

                wechat_qrcode: configData.wechat_qrcode || '',

                alipay_qrcode: configData.alipay_qrcode || ''

            };

            if (configId) {

                await supabase.from('config').update(row).eq('id', configId);

            } else {

                const { data } = await supabase.from('config').insert([row]).select();

                if (data) configId = data[0].id;

            }

            alert('设置已保存');

        }


        function genCarQr() {

            const url = document.getElementById('s-url').value.trim();

            if (!url) { alert('请输入顾客端链接'); return; }

            const c = document.getElementById('car-qrcode'); c.innerHTML = '';

            new QRCode(c, { text: url, width: 200, height: 200, colorDark: '#1C1917', colorLight: '#FFFFFF', correctLevel: QRCode.CorrectLevel.H });

            document.getElementById('car-qr-wrap').classList.remove('hidden');

        }

        const autoUrl = location.href.replace('admin.html', 'index.html');

        document.getElementById('s-url').value = autoUrl;
