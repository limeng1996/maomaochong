            cart.push({ id: Date.now(), name: p.name, emoji: p.emoji, unitPrice: p.price + extra, qty: optionQty, optionsText: opts.join(' | ') });

            updateCartUI(); closeOpt();

            const fab = document.getElementById('cart-fab'); fab.classList.add('scale-110'); setTimeout(() => fab.classList.remove('scale-110'), 200);

        }


        function updateCartUI() {

            const qty = cart.reduce((s, i) => s + i.qty, 0);

            const total = cart.reduce((s, i) => s + i.unitPrice * i.qty, 0);

            document.getElementById('cart-badge').textContent = qty;

            document.getElementById('cart-fab').classList.toggle('hidden', qty === 0);

            document.getElementById('cart-subtotal').textContent = `¥${total}`;

            document.getElementById('cart-total-btn').textContent = `¥${total}`;

            document.getElementById('cart-items').innerHTML = cart.length ? cart.map((it, idx) => `

                <div class="flex gap-3 items-center">

                    <div class="w-12 h-12 bg-stone-100 rounded-lg flex items-center justify-center text-xl">${it.emoji}</div>

                    <div class="flex-1 min-w-0"><h4 class="font-medium text-stone-800 text-sm truncate">${it.name}</h4><p class="text-xs text-stone-400 truncate">${it.optionsText}</p></div>

                    <div class="text-right">

                        <p class="font-bold text-stone-800">¥${it.unitPrice * it.qty}</p>

                        <div class="flex items-center gap-2 mt-1">

                            <button onclick="updQty(${idx},-1)" class="w-6 h-6 rounded-full bg-stone-100 flex items-center justify-center text-stone-600 text-xs press-scale">−</button>

                            <span class="text-sm w-4 text-center">${it.qty}</span>

                            <button onclick="updQty(${idx},1)" class="w-6 h-6 rounded-full bg-stone-100 flex items-center justify-center text-stone-600 text-xs press-scale">+</button>

                        </div>

                    </div>

            `).join('') : '<div class="text-center py-8 text-stone-400">购物车是空的</div>';

        }
